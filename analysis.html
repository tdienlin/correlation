Small Effect

# create data-frame
iq <- seq(100, 110, 1)
income <- seq(1000, 1100, 10)
(d_1 <- data.frame(iq, income))
##     iq income
## 1  100   1000
## 2  101   1010
## 3  102   1020
## 4  103   1030
## 5  104   1040
## 6  105   1050
## 7  106   1060
## 8  107   1070
## 9  108   1080
## 10 109   1090
## 11 110   1100
# calculate correlation
(cor_1 <- cor(d_1$iq, d_1$income))
## [1] 1
# calculate regression coefficient
reg_1 <- lm(income ~ iq, d_1)
summary(reg_1)

# visualization
vis_1 <- ggplot(d_1, aes(iq, income)) +
  geom_point() +
  geom_line() +
  scale_x_continuous(breaks = scales::pretty_breaks(n = 10)) +
  scale_y_continuous(limits = c(0, 2000)) +
  labs(y = "Income (in Dollars)",
       x = "IQ",
       title = "Small Effect",
       subtitle = paste0("r = ", td::my_round(cor_1, "std"), ", b = ", round(reg_1$coefficients["iq"], 0), " $US"))
print(vis_1)
